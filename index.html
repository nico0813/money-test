<script>


const TODAY_CODE = "WAKEN2026";


const questions = [

  {

    q:"Q1 你更舒服的赚钱方式？",

    options:{

      A:"稳稳积累",

      B:"抓爆发机会",

      C:"提升技能赚钱",

      D:"靠合作放大"

    }

  },

  {

    q:"Q2 面对风险你更像？",

    options:{

      A:"保守优先",

      B:"敢冲",

      C:"分析优化",

      D:"找人一起"

    }

  }

];


const types = {

  A:{

    name:"稳健累积型",

    desc:"你适合长期系统积累，慢但稳。2026关键词：纪律、复利、规则。"

  },

  B:{

    name:"进攻成长型",

    desc:"你适合抓窗口期。2026关键词：节奏、仓位、风险控制。"

  },

  C:{

    name:"认知套利型",

    desc:"你适合提升单价赚钱。2026关键词：技能变现、输出。"

  },

  D:{

    name:"资源整合型",

    desc:"你适合项目和合作。2026关键词：杠杆、规则。"

  }

};


let answers = [];


function unlock(){

  const input = document.getElementById("code").value.trim();

  if(input !== TODAY_CODE){

    alert("口令错误");

    return;

  }

  startQuiz();

}


function startQuiz(){

  document.getElementById("quiz").classList.remove("hidden");

  renderQuestion(0);

}


function renderQuestion(index){

  if(index >= questions.length){

    showResult();

    return;

  }


  const quiz = document.getElementById("quiz");

  quiz.innerHTML = "<h3>"+questions[index].q+"</h3>";


  for(let key in questions[index].options){

    const div = document.createElement("div");

    div.className = "option";

    div.innerText = key+"："+questions[index].options[key];

    div.onclick = function(){

      answers.push(key);

      renderQuestion(index+1);

    };

    quiz.appendChild(div);

  }

}


function showResult(){

  document.getElementById("quiz").classList.add("hidden");

  document.getElementById("resultBox").classList.remove("hidden");


  const score = {A:0,B:0,C:0,D:0};

  answers.forEach(a=>score[a]++);


  let maxType = "A";

  for(let key in score){

    if(score[key] > score[maxType]){

      maxType = key;

    }

  }


  document.getElementById("resultText").innerText = "你的钱味体质："+types[maxType].name;

  document.getElementById("resultDesc").innerText = types[maxType].desc;

}
